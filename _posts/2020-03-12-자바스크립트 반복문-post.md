* while - 지정된 조건이 참이면 반복 실행
```html
<script>
    var i = 0;

    // i가 10보다 작으면 반복실행
    while (i<10) {
        document.write(i+'<br>');
        i++;
    }
</script>
```

<br>

* for - 초기식을 실행한 후에 조건식의 값이 참인 동안 코드를 실행
```html
<!-- 
for(초기식; 조건식; 증감식) {
    반복할 코드;
}
 -->
<script>
    for (i=0; i<10; i++) {
        document.write(i+'<br>');
    }
</script>
```

### 온도변환기 예제
```html
<html>
    <head>
        <title>온도 변환기</title>
    </head>

    <body>
        <table border="2">
            <tr>
                <td>섭씨</td>
                <td>화씨</td>
            </tr>
        
            <script>
                for (celsius=0; celsius<=10; celsius++) {
                    document.write('<tr><td>'+celsius+'</td><td>'+((celsius*9.0/5)+32)+'</td></tr>');
                }
            </script>
        </table>
    </body>
</html>
```

### 구구단 예제
```html
<script>
    document.write('<h1>구구단</h1>');
    document.write('<table border="2">');

    for (var i=1; i<=9; i++) {
        document.write('<tr>');
        document.write('<td>'+i+'</td>');

        for (var j=2; j<=9; j++) {
            document.write('<td>'+i*j+'</td>');
        }

        document.write('</tr>');
    }

    document.write('</table>');
</script>
```
<br>

* do-while - 일단 문장을 한 번 실행하고 나서 조건을 검사하고 싶을 때 사용
```html
<script>
    var i=0;

    do {
        document.write(i+'<br>');
        i++;
    } while (i<10);
</script>
```

* for/in
```html
<script>
    var myCar = {
        model : "bmw",
        color : "gray",
        hp : 100
    };
    var txt = '';

    for (var x in myCar) {
        txt += myCar[x] + ' ';
    }

    document.write(txt);
</script>
```


<br>

* break
```html
<script>
    var msg = '';
    for (var i=0; i<10; i++) {
        if (i==3) break; // i가 3이면 반복을 중단하고 반복 루프를 빠져나감
        msg += i + '<br>';
    }
    document.write(msg);
</script>
```
<br>

* continue
```html
<script>
    var msg = '';
    for (var i=0; i<10; i++) {
        if (i==3) continue; // i가 3이면 반복 중단하고 다음 반복을 시작한다.
        msg += i + '<br>';
    }
    document.write(msg);
</script>
```
